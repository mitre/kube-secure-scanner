    Mermaid Diagram (WCAG Compliant)¶

    flowchart TD
    start([Start]) --> step1

    subgraph step1["STEP 1: DEPLOY POD WITH SIDECAR"]
        deploy_target["Deploy Target Container in Kubernetes Pod"] --> deploy_sidecar["Deploy Scanner Sidecar Container in Same Pod"]
        deploy_sidecar --> shared_namespace["Enable Shared Process Namespace Between Containers"]
    end

    step1 --> step2

    subgraph step2["STEP 2: PERFORM SCAN USING SIDECAR"]
        find_process["Sidecar Finds Target Process Using 'ps'"] --> access_fs["Access Target Filesystem via /proc/PID/root"]
        access_fs --> run_cinc["Run CINC Auditor Against Target Container's Filesystem"]
        run_cinc --> store_results["Store Results in Shared Volume"]
        store_results --> process_results["Process Results with SAF CLI & Validate Against Threshold"]
        process_results --> retrieve_results["Retrieve Results from Sidecar via kubectl cp or Volume Mount"]
    end

    step2 --> complete([Scan Complete])
